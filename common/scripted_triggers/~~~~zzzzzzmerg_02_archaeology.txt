default_site_visible_trigger = {
	OR = {
		merg_is_default_empire = yes
		AND = {
			merg_has_wula_fe = yes
			is_fallen_empire = yes
		}
	}
	OR = {
		intel_level = {
			level >= low
			system = from.solar_system
		}
		AND = {
			exists = from
			from.planet = { is_within_borders_of = root }
		}
	}
}
